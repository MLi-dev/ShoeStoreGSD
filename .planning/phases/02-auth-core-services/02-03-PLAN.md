auth_service.py public API:
```python
def register(email: str, password: str) -> dict
# Success: {"success": True, "data": {"user_id": str, "email": str}}
# Duplicate: {"success": False, "code": "EMAIL_ALREADY_EXISTS", ...}

def login(email: str, password: str) -> dict
# Success: {"success": True, "data": {"token": str, "user_id": str}}
# Fail: {"success": False, "code": "INVALID_CREDENTIALS", ...}

def verify_token(token: str) -> dict
# Success: {"success": True, "data": {"user_id": str}}
# Expired: {"success": False, "code": "TOKEN_EXPIRED", ...}
# Invalid: {"success": False, "code": "INVALID_TOKEN", ...}

def reset_request(email: str) -> dict
# Always success; token in data["token"] if email exists, else None

def reset_confirm(token: str, new_password: str) -> dict
# Success: {"success": True, "data": {"message": "Password updated successfully"}}
# Invalid/expired token: {"success": False, "code": "INVALID_RESET_TOKEN", ...}
```

Helper to seed an order at a given status:
```python
from datetime import datetime, timezone
from uuid import uuid4
from app.lib.orders.models import Order, OrderItem
from app.lib.orders.store import orders_db

def _seed_order(order_id=None, user_id="u1", order_status="placed", payment_status="pending"):
    oid = order_id or str(uuid4())
    now = datetime.now(timezone.utc).isoformat()
    o = Order(id=oid, user_id=user_id, items=[OrderItem(product_id="p1", quantity=1, unit_price=99.0)],
              total_amount=99.0, payment_method="credit_card", payment_status=payment_status,
              order_status=order_status, created_at=now, updated_at=now)
    orders_db[oid] = o
    return o
```
