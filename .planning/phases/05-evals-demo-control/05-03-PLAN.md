```python
# tests/evals/test_smoke.py
# Smoke runner: one real Anthropic API call per eval dataset (3 total).
# Skips all tests when ANTHROPIC_API_KEY is absent — safe for CI with no key.
# Source: CONTEXT.md D-12, D-13
import logging
import os

import pytest

from app.lib.agent import agent
from app.lib.agent import history as agent_history
from app.lib.auth.store import reset_tokens_db, users_db
from app.lib.cart.store import carts_db
from app.lib.catalog.store import products_db
from app.lib.evals.datasets.adversarial import CASES as ADVERSARIAL_CASES
from app.lib.evals.datasets.negative import CASES as NEGATIVE_CASES
from app.lib.evals.datasets.positive import CASES as POSITIVE_CASES
from app.lib.orders.store import orders_db
from app.lib.seed.seed import seed

logger = logging.getLogger(__name__)

# Skip entire module when API key is absent — not a failure, just not runnable.
pytestmark = pytest.mark.skipif(
    not os.environ.get("ANTHROPIC_API_KEY"),
    reason="ANTHROPIC_API_KEY not set — skipping live eval smoke run",
)


@pytest.fixture(autouse=True)
def reset_stores():
    """Reset all in-memory stores and re-seed before each smoke test."""
    users_db.clear()
    reset_tokens_db.clear()
    carts_db.clear()
    orders_db.clear()
    products_db.clear()
    agent_history._history.clear()
    seed()
    yield
    users_db.clear()
    reset_tokens_db.clear()
    carts_db.clear()
    orders_db.clear()
    products_db.clear()
    agent_history._history.clear()


@pytest.fixture
def smoke_user():
    """Return the seeded alice User object for agent.run() calls."""
    return next(u for u in users_db.values() if u.email == "alice@example.com")


@pytest.mark.asyncio
async def test_smoke_positive(smoke_user):
    """Smoke: one positive case — happy-path agent call must not raise."""
    case = POSITIVE_CASES[0]
    logger.info(
        "Smoke positive: tags=%r input=%r",
        case["tags"],
        case["input"],
    )
    result = await agent.run(
        user_id=smoke_user.id,
        user=smoke_user,
        message=case["input"],
    )
    assert "success" in result, f"agent.run returned no 'success' key: {result!r}"


@pytest.mark.asyncio
async def test_smoke_negative(smoke_user):
    """Smoke: one negative case — error-scenario agent call must not raise."""
    case = NEGATIVE_CASES[0]
    logger.info(
        "Smoke negative: tags=%r input=%r",
        case["tags"],
        case["input"],
    )
    result = await agent.run(
        user_id=smoke_user.id,
        user=smoke_user,
        message=case["input"],
    )
    assert "success" in result, f"agent.run returned no 'success' key: {result!r}"


@pytest.mark.asyncio
async def test_smoke_adversarial(smoke_user):
    """Smoke: one adversarial case — hostile-input agent call must not raise."""
    case = ADVERSARIAL_CASES[0]
    logger.info(
        "Smoke adversarial: tags=%r input=%r",
        case["tags"],
        case["input"],
    )
    result = await agent.run(
        user_id=smoke_user.id,
        user=smoke_user,
        message=case["input"],
    )
    assert "success" in result, f"agent.run returned no 'success' key: {result!r}"
```

**Implementation notes:**
- The `pytestmark` with `skipif` applies to the whole module — all 3 tests skip together when no API key.
- `agent_history._history` is the module-level dict in `history.py` — access it directly to clear between tests.
- `smoke_user` fixture depends on `reset_stores` fixture having already run (autouse=True ensures order).
- Do NOT validate `expected_trajectory` or `expected_output` — smoke run only checks no uncaught exception (D-13).
- Check pyproject.toml for `asyncio_mode`. If it is "auto", no `@pytest.mark.asyncio` decorator is needed on individual tests (but adding it is harmless). If it is not set, add `asyncio_mode = "auto"` under `[tool.pytest.ini_options]`.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Smoke runner → Anthropic API | Real HTTPS calls using ANTHROPIC_API_KEY from environment |
| Smoke runner → in-memory stores | Fixture resets all stores then calls seed() — no shared state between tests |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-09 | Information Disclosure | ANTHROPIC_API_KEY in test environment | mitigate | Key read from environment variable only — never hardcoded or logged. `pytest --collect-only` path (no API key) never touches the key. |
| T-05-10 | Denial of Service | Smoke test makes real API calls — cost and rate limit risk | accept | Only 3 calls per run (one per dataset); guarded by `skipif` so CI without key never incurs cost. Acceptable for a personal learning project. |
| T-05-11 | Tampering | reset_stores fixture clears and re-seeds between tests | accept | Intentional — isolation by design. Only test-local state; no production data at risk. |
| T-05-12 | Elevation of Privilege | Adversarial case input reaches real Anthropic API | accept | The adversarial inputs (injection strings, off-topic requests) test that the agent's guardrails work correctly. The agent's system prompt and guardrails are the defense layer — this is the point of the test. |
